            _log[name] += value
        else
            _multilog[name] = 1
            _log[name] = list( _log[name], value )

        _log[name] = value

        _LOGADD("expected", actual)

    var/value

        name = "log"
        value = arg1
    else
        name = arg1
        value = arg2
    _LOGADD(name, value)
