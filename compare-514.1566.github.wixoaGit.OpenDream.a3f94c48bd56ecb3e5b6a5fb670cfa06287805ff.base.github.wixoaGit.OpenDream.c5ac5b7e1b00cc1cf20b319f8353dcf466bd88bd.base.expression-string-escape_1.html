        if (name in _multilog)
            _log[name] += value
        else
            _multilog[name] = 1
            _log[name] = list( _log[name], value )
    else
        _log[name] = value

    if (actual ~= expected) 
        _LOGADD("expected", actual)
    else
        _mismatch[name] = list(actual, expected)

    var/value

        value = arg1
